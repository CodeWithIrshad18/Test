SELECT
    TRBDGDA_BD_PNO,
    SUM(CASE 
            WHEN TRBDGDA_BD_DATE >= '2025-11-01' 
             AND TRBDGDA_BD_DATE <  '2025-12-01' 
            THEN 1 ELSE 0 
        END) AS November_Punches,

    SUM(CASE 
            WHEN TRBDGDA_BD_DATE >= '2025-12-01' 
             AND TRBDGDA_BD_DATE <  '2026-01-01' 
            THEN 1 ELSE 0 
        END) AS December_Punches,

    SUM(CASE 
            WHEN TRBDGDA_BD_DATE >= '2026-01-01' 
             AND TRBDGDA_BD_DATE <  '2026-02-01' 
            THEN 1 ELSE 0 
        END) AS January_Punches
FROM T_TRBDGDAT_EARS
WHERE TRBDGDA_BD_ENTRYUID = 'MOBILE'
  AND TRBDGDA_BD_DATE >= '2025-11-01'
  AND TRBDGDA_BD_DATE <  '2026-02-01'
GROUP BY TRBDGDA_BD_PNO
ORDER BY TRBDGDA_BD_PNO;
